01. 결괏값을 작성하시오.

function q2() {
    const array = [10, 31, 1, 2, 3 ,4, 32];
    return array.sort();
}

console.log(q2());
결과보기
1 10 2 3 31 32 4
해설
sort( )는 순서대로 정렬하는 메서드로 함수가 포함되지 않으면 유니코드 값에 따라 정렬됨
따라서 1 2 3 같은 숫자 순이 아닌 앞자리가 가장 빠른 순인 1 10 2 3 순이 되는 것

02. 결괏값을 작성하시오.

function q3() {
    const array = [10, 31, 1, 2, 3 ,4, 32];
    return array.filter(x => String(x).length >= 2);
}

console.log(q3());
결과보기
10 31 32
해설
String( )는 배열값을 문자열로 만들어주는 메서드
숫자가 문자열이 되었으므로 각 문자열의 length는 길이값이 됨
따라서 길이값 2 이상인 2자리 수가 정답

03. 결괏값을 작성하시오.

function q7() {
    const str = "taraeaea";
    return str.split('').map(word => {
        if(word !== "a") return word;
    }).join('');
}

console.log(q7());
결과보기
tree
해설
split( )으로 문자열을 각각 하나의 문자열로 분해하고 "a"가 아닌 문자열을 반환하는 문제
return은 반환이자 종료의 의미를 갖지만 각각의 배열 문자열에 적용되므로 t 하나로 끝나지 않음
또한 a의 경우 undefined가 반환되지만 join( )에 의해 생략이 되버림

function q8() {
    let answer = 0;
    const array = [1,2,3,4,5];
    for(let i = 3; i < 8; i++) {
        if(array.includes(i) && 0) {
            answer++;
        }
    }
    return answer;
}

console.log(q8());
결과보기
0
해설
include의 반환값을 물어보는 문제
해당되는 값을 반환하면 true가 되고 없으면 false를 반환하는데 if문에서 true면 통과가 되는 식
하지만 && 0 에 의해 무조건 false가 포함되는 함정이 있어서 답은 0이됨

05. 결괏값을 작성하시오.

function q10(arr) {
    let flag = false;
    let n = 1;
    while(!flag) {
        n++;
        for(let i = 1; i < arr.length; i++) {
            if((arr[0] * n) % arr[i] === 0) {
                flag = true;
            }
            else {
                flag = false;
                break;
            }
        }
    }
    return arr[0] * n
}

console.log(q10([2,6,8,14]));
결과보기
168
해설
6,8,14의 최소공배수를 구하는 문제
while문의 flag가 false면 반복 실행되고 결국 if문이 전부 true가 되어야 함수가 중지되고 값이 반환됨
2*n의 값을 6과 8과 14로 나누어 나머지가 0이 되는 최소값은 168이므로 답이 168이됨
