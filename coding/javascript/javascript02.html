01. 변수 : 데이터 불러오기

"let" 명령문은 var의 진화형으로 변수를 선언하며, 변수가 연속될 떄 첫 let를 선언 후 생략이 가능합니다.
var과의 차이점은 중복 선언이 불가능 하며, 에러 메세지가 출력된다는 점에 있습니다. 

let x = 100,        // 중복되는 let를 생략 가능
         y = 200, 
         z = "javascript";
    
    document.write(x);
    document.write(y);
    document.write(z);
결과보기
100
200
"javascript"

06. 배열 : 데이터 불러오기 : for()문

for 반복문은 거짓이 판명될 때까지 반복합니다. 초기화를 통해 변수를 초기화하며, 조건문을 통해 매 반복마다 참과 거짓을 구별하고 증감식을 통해 반복시 카운터 변수를 증감합니다. 

const arr = [100, 200, 300, 400, 500];
    
    //document.write(arr[0]);
    //document.write(arr[1]);
    //document.write(arr[2]);
    //document.write(arr[3]);
    //document.write(arr[4]);
    
    //for(초기값, 조건식, 증감식)
    for(let i=0; i<arr.length; i++){   //출력 반복적으로 쓰는걸 줄이는 방식. ++는 하나씩 순서대로 표현해주는 방식  
        document.write(arr[i]);     // i=0, i<9, document, i++ 순으로 돌아가며 조건이 맞지 않을때까지 반복된다.
    }

07. 배열 : 데이터 불러오기 : forEach()

forEach 반복문은 요소들을 반복하여 작업을 수행할 때 사용합니다.
for문을 사용할 때 길어지는 코드를 forEach 반목문을 사용하면 간단하게 줄일 수 있습니다. 

const num = [100, 200, 300, 400, 500];
    
    //for문을 이용해서 출력
    for(let i=0; i<num.length; i++){
        document.write(num[i]);
    };
    
    //forEach()를 이용해서 출력. forEach는 매쏘드
    num.forEach(function(el){         //실행문 안에 실행문이 들어간걸 콜백함수 라고 함. element는 요소나 인자로 el로 줄여서 씀
        document.write(el);
    });
    
    //forEach() 3가지 인자()값
    num.forEach(function(element, index, array){    // function에 여러 요소가 들어갈 수 있다.
        document.write(element);
        document.write(index);
        document.write(array);       //array는 출력값에 "," 가 들어가는 특징이 있다.
    })

08. 배열 : 배열에 데이터 불러오기 : for of (요소값을 출력함)

for of 반복문은 요소의 값을 반복하여 출력하기 위해 사용하며 forEach 반복문의 형태를 조금 더 간단하게 줄일 수 있습니다. 

const arr = [100, 200, 300, 400 ,500];
    
    for( let i=0; i<arr.length; i++){
        document.write(arr[i]);
    }
    
    document.write("<br>")
    
    for( let i of arr ){        //for문을 forEach문으로 바꿔서 더 간단하게 만듬
        document.write(i);
    }

19. 객체 : 데이터 불러오기 : 펼침연산자 - 복사

펼침연산자는 점3개(...)로 표시하며 복잡한 갹체 나열을 단순화하거나 객체의 병합, 함수의 인자 처리를 단순하게 처리할 수 있도록 개선하는 함수입니다.

{
    const obj = {
        a: 100,
        b: 200,
        c: "javascript"
    }

    //복사
    const spread = { ...obj }   //...이 펼침연산자의 기능

    document.write(spread.b);
    document.write(spread.b);
    document.write(spread.b);
}

22. 객체 : 데이터 불러오기 : 비구조화 할당

23. 객체 : 데이터 불러오기 : 객체구조분해할당
